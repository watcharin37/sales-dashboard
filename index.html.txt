    <script>
        function togglePhase(phaseId) {
            const content = document.getElementById(phaseId);
            const header = content.previousElementSibling;
            const isActive = content.classList.contains('active');

            document.querySelectorAll('.phase-content.active').forEach(activeContent => {
                activeContent.classList.remove('active');
                activeContent.previousElementSibling.classList.remove('active');
            });

            if (!isActive) {
                content.classList.add('active');
                header.classList.add('active');
            }
        }

        function toggleTask(taskElement, event) {
            event.stopPropagation(); 
            const checkbox = taskElement.querySelector('.checkbox');
            checkbox.checked = !checkbox.checked;
            taskElement.classList.toggle('completed', checkbox.checked);
            updateProgress();
        }
        
        document.querySelectorAll('.task-list li').forEach(li => {
            li.addEventListener('click', function(event) {
                toggleTask(this, event);
            });
        });

        function updateProgress() {
            const totalTasks = document.querySelectorAll('.task-list li').length;
            if (totalTasks === 0) {
                document.getElementById('progressText').textContent = 'ไม่มี Task ในระบบ';
                return;
            };
            
            const completedTasks = document.querySelectorAll('.task-list li.completed').length;
            const progressPercent = Math.round((completedTasks / totalTasks) * 100);
            
            const progressBarFill = document.getElementById('progressBar');
            progressBarFill.style.width = progressPercent + '%';
            progressBarFill.textContent = progressPercent > 10 ? `${progressPercent}%` : '';

            document.getElementById('progressText').textContent = `${progressPercent}% สำเร็จ (${completedTasks}/${totalTasks} Tasks)`;
        }

        function addProspect() {
            const name = prompt('ชื่อ Prospect:');
            if (name) {
                const contact = prompt('ข้อมูลติดต่อ:');
                const status = prompt('สถานะ (Cold/Warm/Hot):');
                alert(`✅ เพิ่ม Prospect สำเร็จ!\n\nชื่อ: ${name}\nติดต่อ: ${contact}\nสถานะ: ${status}`);
            }
        }

        function emergencyMode() {
            if (confirm('🚨 เปิดโหมดฉุกเฉิน?\n\nจะปรับเป้าหมายเป็น 10 ชั่วโมง/วัน และเพิ่มจำนวน prospects')) {
                alert('🔥 โหมดฉุกเฉิน เปิดแล้ว!\n\n✅ เป้าหมายรายวัน: 10 ชั่วโมง\n✅ Prospects: 15 ราย/สัปดาห์\n✅ Follow-up: ทุกวัน');
            }
        }

        document.addEventListener('DOMContentLoaded', function() {
            updateProgress();
            togglePhase('phase1');
        });
    </script>
